func WarshallFloyd(wf [][]int) {
	n := len(wf)
	for k := 0; k < n; k++ {
		for i := 0; i < n; i++ {
			for j := 0; j < n; j++ {
				wf[i][j] = min(wf[i][j], wf[i][k]+wf[k][j])
			}
		}
	}
}
func WarshallFloydAppendEdge(from, to, cost int, wf [][]int) {
	n := len(wf)
	for i := 0; i < n; i++ {
		for j := 0; j < n; j++ {
			wf[i][j] = min(wf[i][j], wf[i][from]+wf[to][j]+cost)
		}
	}
}
